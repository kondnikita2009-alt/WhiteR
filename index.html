import random

# INPUT
n = int(input("Count (>6): "))
min_v = int(input("Min: "))
max_v = int(input("Max: "))

data = [random.randint(min_v, max_v) for _ in range(n)]
print(f"Data: {data}")

# TARGETS
targets = [(1, "2nd"), (3, "4th"), (5, "6th")]

for idx, name in sorted(targets, reverse=True):
    if idx < len(data):
        val = data.pop(idx)
        print(f"Terminated {name}: {val}")

print(f"Result: {data}")

bad_vals = list(map(int, input("To remove: ").split()))

for v in bad_vals:
    while v in nums:
        nums.remove(v)

if not nums: print("VOID")
else: print(f"Clean: {nums}")
